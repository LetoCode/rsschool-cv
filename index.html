# требуется создать спираль из цифр
# начало - в центре
# максимально 72 строки кода

from math import ceil


m_size = 10
i = 2 
z = m_size**2
x, y = int(m_size/2-1), int(m_size/2-1)
d = {1: [x, y]}
inter_iter = 1

def step_to(p, way, inter_iter):
  global i, y, x, z
  while p <= inter_iter and i <= z :
    if way == 'right':
      y = y + 1
    elif way == 'down':
      x = x + 1
    elif way == 'left':
      y = y - 1
    elif way == 'up':
      x = x - 1

    d[i] = [x, y]
    i +=1; p +=1

while i <= z:
  p1 = p2 = p3 = p4 = 1

  step_to(p1, 'right', inter_iter)
  step_to(p1, 'down', inter_iter)
  inter_iter +=1

  step_to(p1, 'left', inter_iter)
  step_to(p1, 'up', inter_iter)
  inter_iter +=1

matrix = []

matrix = [['_' for m in range(m_size)] for m1 in range(m_size)]

for key, value in d.items():
  matrix[value[0]][value[1]] = key

with open('matrix.txt', 'w') as f:
  for i in matrix:
    for j in i:
      f.write(('{: >' + str(len(str(z))+1) + '}').format(j)) 
    f.write('\n')
